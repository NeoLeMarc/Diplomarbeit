\section{Analyse}
\subsection{Anforderungen}

\begin{frame}
    \frametitle{Anforderungen}

    \begin{itemize}
        \item Empfangen von Vitaldaten (Puls \& Atmung) von Sensoren
        \item Senden von Befehlen (z.B. Alarm aus, manueller Alarm) an Sensoren
        \item Schnelle Einsatzfähigkeit: Je weniger Kabel desto besser.
        \item Skalierbar in Anzahl der Patienten {\bf und} Helfern
        \item Kostengünstig
        \item Niedriger Stromverbrauch
        \item Robust 
        \item Einfache Bedienbarkeit
        \item Erweiterbar
    \end{itemize}
\end{frame}


\subsection{Stand der Technik}

\begin{frame}
    \frametitle{Stand der Technik}

    \begin{itemize}
        \item Tragbare EKG-Geräte, insbesondere in Defis\\
              $\rightarrow$ Schwer, groß, teuer 
        \item Monitore mit Bluetooth-Technologie\\
              $\rightarrow{}$ Maximal 8 aktive Geräte pro Netzwerk
        \item Monitore mit WLAN-Technologie\\ 
              $\rightarrow{}$ störungsanfällig, hoher Stromverbrauch, teuer, kein Roaming.
        \item Herstellerspezifische Lösungen (z.B. Z-Wave oder Polar)\\
              $\rightarrow{}$ Proprietär, teuer
        \item $I^2C$ basiertes Sensornetz\\
              $\rightarrow{}$ Benötigt viele Kabel
    \end{itemize}


\end{frame}

                \item Mesh-Netzwerk: Durch Router erweiterbar

\section{Fazit}

\begin{frame}
    \frametitle{Verbesserungen gegenüber dem Stand der Technik}
    \begin{itemize}
        \item Keine Kabel mehr notwendig
        \item Kostengünstiger als alle existierenden Lösungen
        \item Stromsparender als alle existierenden Lösungen
        \item Mehr Teilnehmer als in allen existierenden Lösungen
        \item Hohe Reichweite durch Einsatz von Repeatern erreichbar
        \item Flexible und Erweiterbare Software 
        \item Herstellerunabhängig: Hardware-/Softwareschnittstelle austauschbar 
        \item Unabhängig von lokaler Infrastruktur
    \end{itemize}
\end{frame}
